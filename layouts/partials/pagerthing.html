{{/* The following pagination based heavily on code derived from: https://glennmccomb.com/articles/how-to-build-custom-hugo-pagination/ */}} {{/* // // OUTPUT POSTS //–––––––––––––––––––––––––––––––––––––––––––––––––– */}} {{ $paginator := .Paginate (where .Data.Pages "Type" "posts") }} {{ $paginator := .Paginator }} {{/* Number of links either side of the current page. */}} {{ $adjacent_links := 2 }} {{/* $max_links = ($adjacent_links * 2) + 1 */}} {{ $max_links := (add (mul $adjacent_links 2) 1) }} {{/* $lower_limit = $adjacent_links + 1 */}} {{ $lower_limit := (add $adjacent_links 1) }} {{/* $upper_limit = $paginator.TotalPages - $adjacent_links */}} {{ $upper_limit := (math.Sub $paginator.TotalPages $adjacent_links) }}
{{- /* First page --- test for both page 1 and page 2 to avoid redundant links: https://webaim.org/discussion/mail_thread?thread=9881 https://www.boia.org/blog/how-redundant-links-impact-accessibility https://stackoverflow.com/questions/38754694/accessibility-wcag-2-0-redundant-links-and-responsive-website-navigation */ -}}
{{ if and (ne $paginator.PageNumber 1) (ne $paginator.PageNumber 2) }} {{ partials.IncludeCached "svgPrevPageIcon.html" . }}{{ partials.IncludeCached "svgPrevPageIcon.html" . }} {{ else }} {{ partials.IncludeCached "svgFirstPageIcon.html" . }}{{ partials.IncludeCached "svgFirstPageIcon.html" . }} {{ end }}
{{- /* Previous page. */ -}}
{{ if $paginator.HasPrev }} {{ partials.IncludeCached "svgPrevPageIcon.html" . }} {{ else }} {{ partials.IncludeCached "svgFirstPageIcon.html" . }} {{ end }}
{{- /* Next page. */ -}}
{{ if $paginator.HasNext }} {{ partials.IncludeCached "svgNextPageIcon.html" . }} {{ else }} {{ partials.IncludeCached "svgLastPageIcon.html" . }} {{ end }}
{{- /* Last page --- test for both last page and next-to-last page to avoid redundant links: https://webaim.org/discussion/mail_thread?thread=9881 https://www.boia.org/blog/how-redundant-links-impact-accessibility https://stackoverflow.com/questions/38754694/accessibility-wcag-2-0-redundant-links-and-responsive-website-navigation */ -}}
{{ if (and ( (ne $paginator.PageNumber $paginator.TotalPages) ) (ne $paginator.PageNumber (math.Sub $paginator.TotalPages 1))) }} {{ partials.IncludeCached "svgNextPageIcon.html" . }}{{ partials.IncludeCached "svgNextPageIcon.html" . }} {{ else }} {{ partials.IncludeCached "svgLastPageIcon.html" . }}{{ partials.IncludeCached "svgLastPageIcon.html" . }} {{ end }}
