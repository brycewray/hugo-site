{{/* Detect imgs and generate CSS classes for them. Requires using page bundles. Based on comments in: https://github.com/danielfdickinson/image-handling-mod-hugo-dfd/pull/72 */}} {{- with .Resources.ByType "image" -}} {{- range . -}} {{- $src := . -}} {{- $imgBd5 := md5 .Name -}} {{- $GIP_colors := $src.Colors -}} {{- if (lt ($GIP_colors | len) 2) -}} {{- $GIP_colors = $GIP_colors | append "#000000" -}} {{- end -}} {{- $GIP_bkgd := delimit ($GIP_colors) ", " -}} {{- $BkgdStyleGIP := print "background: linear-gradient(" $GIP_bkgd "); background-size: cover; background-repeat: no-repeat;" -}} {{- $LQIP_img := $src.Resize "20x jpg q20" -}} {{- $LQIP_b64 := $LQIP_img.Content | base64Encode -}} {{- $BkgdStyleLQIP := print "background: url(data:image/jpeg;base64," $LQIP_b64 "); background-size: cover; background-repeat: no-repeat;" -}} {{- end -}} {{- end -}}
